# Input from the user
a = float(input("Enter the first number (a): "))
b = float(input("Enter the second number (b): "))

# 1. Floor Division
floor_division_result = a // b

# 2. a power b
power_result = a ** b

# 3. Check if two variables are equal
are_equal = a == b

# Print the results
print("1. Floor Division:", floor_division_result)
print("2. a power b:", power_result)
print("3. Are a and b equal?", are_equal)
